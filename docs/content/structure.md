```
├── Parvula        // Source files
├── data           // Data files
│   ├── config     // Parvula configuration files
│   ├── logs       // Log files (@next version)
│   ├── pages      // Pages
│   └── users      // Users
├── plugins        // Contains your plugins
├── static         // Static files [public]
│   ├── files
│   └── ...
├── themes         // Contains themes [public]
└── vendor         // Parvula and plugin dependencies (created by composer)
```
